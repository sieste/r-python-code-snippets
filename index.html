Filtering Data

Selecting rows and columns based on conditions and creating subsets.

library(dplyr)

# Filter rows where age > 30
filtered <- data %>%
  filter(age > 30)

# Select specific columns
subset <- data %>%
  select(name, age, salary)

# Chain operations
result <- data %>%
  filter(department == "Sales") %>%
  select(-email)
import pandas as pd

# Filter rows where age > 30
filtered = data[data['age'] > 30]

# Select specific columns
subset = data[['name', 'age', 'salary']]

# Chain operations
result = data[data['department'] == 'Sales']\
    .drop(columns=['email'])

# Alternative with loc
result = data.loc[
    data['department'] == 'Sales',
    data.columns != 'email'
]

Grouping and Aggregation

Computing statistics grouped by categorical variables.

library(dplyr)

# Group by department and calculate mean
summary_stats <- data %>%
  group_by(department) %>%
  summarise(
    avg_salary = mean(salary),
    count = n(),
    max_age = max(age)
  )

# Multiple aggregations
agg <- data %>%
  group_by(region) %>%
  summarise(
    total = sum(sales),
    avg = mean(sales)
  )
import pandas as pd

# Group by department and calculate mean
summary_stats = data.groupby('department').agg({
    'salary': 'mean',
    'age': 'max'
}).rename(columns={'salary': 'avg_salary'})
summary_stats['count'] = data.groupby(
    'department').size()

# Multiple aggregations
agg = data.groupby('region').agg({
    'sales': ['sum', 'mean']
}).rename(
    columns={'sum': 'total', 'mean': 'avg'}
)

Merging Datasets

Combining multiple dataframes using join operations.

library(dplyr)

# Inner join
combined <- left_join(
  employees, departments,
  by = "dept_id"
)

# Join with renamed key
result <- inner_join(
  df1, df2,
  by = c("employee_id" = "emp_id")
)

# Multiple keys
merged <- full_join(
  sales, regions,
  by = c("year", "region")
)
import pandas as pd

# Inner join
combined = employees.merge(
    departments,
    left_on='dept_id',
    right_on='dept_id',
    how='left'
)

# Join with renamed key
result = df1.merge(
    df2,
    left_on='employee_id',
    right_on='emp_id',
    how='inner'
)

# Multiple keys
merged = sales.merge(
    regions,
    on=['year', 'region'],
    how='outer'
)
